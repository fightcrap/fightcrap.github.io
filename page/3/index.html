    private int serch(int[] nums, int start, int end, int target) {

    if (start >= end) {
    return start;
}

int mod = (start + end) / 2;
if (nums[mod] == target) {
    return mod;
}
if (nums[mod] < target) {
    return serch(nums, mod + 1, end, target);
}
